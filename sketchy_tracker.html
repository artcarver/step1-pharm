// ── DATA let watchedState = []; const GIST_ID = 'your_gist_id_here'; // replace with your actual Gist ID const GIST_FILENAME = 'watched_state.json'; // name of the file in your Gist async function loadWatched() { try { const response = await fetch(`https://api.github.com/gists/${GIST_ID}`); const gist = await response.json(); const fileContent = gist.files[GIST_FILENAME].content; watchedState = JSON.parse(fileContent); } catch (error) { console.error('Error loading watched state:', error); } } async function saveWatched() { try { const jsonData = JSON.stringify(watchedState); await fetch(`https://api.github.com/gists/${GIST_ID}`, { method: 'PATCH', headers: { 'Content-Type': 'application/json', 'Authorization': 'token your_personal_access_token_here' // add your token }, body: JSON.stringify({ files: { [GIST_FILENAME]: { content: jsonData } } }) }); } catch (error) { console.error('Error saving watched state:', error); } }
